# Welcome to PSM Design System

<div className="subheading">Installation</div>

```
// NPM
npm i @psm/components

// Yarn
yarn add @psm/components
```

<div className="subheading">Configuring Theme Provider</div>

```
import React from "react";
import { ThemeProvider, theme } from "@psm/theme";
import Home from "./Home";

const App = () => {
  return (
    <ThemeProvider theme={theme}>
      <Home />
    </ThemeProvider>
  );
};

export default App;
```

<div className="subheading">Using Emotion Styled</div>

```
import React from "react";
import styled from "@emotion/styled";
import { Button } from "@psm/components";

const StyledButton = styled(Button)`
  background-color: red;
`;

export const MyApp: React.FC = () => (
  <StyledButton>Button Label</StyledButton>
);
```
